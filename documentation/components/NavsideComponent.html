import { Component, ElementRef, OnInit, ViewChild } from '@angular/core';
import { DataService } from '../services/data.service';
import { OptionChartService } from '../services/option-chart.service';
import { EChartsOption } from 'echarts';

@Component({
	selector: 'app-navside',
	templateUrl: './navside.component.html',
	styleUrls: ['./navside.component.css']
})
export class NavsideComponent implements OnInit {

	contributors: any = []
	books: any = []
	options: any
	optionPie: EChartsOption = {};
	optionBar: EChartsOption = {};
	optionPolarBar: EChartsOption = {};
	optionLine: EChartsOption = {};
	optionRadar: EChartsOption = {};
	optionPolarRadialBar: EChartsOption = {};

	types = ['line', 'bar', 'pie', 'scatter', 'effectScatter', 'candlestick', 'funnel']
	type = 'bar'
	bc: any = []
	key: any
	title: string
	data:any = []
	data1: any = []
	data2: any = []
	dataObject: any = []
	value:string

	/**
	 * 
	 * @param dataService est service pour récupérer toutes les données de la BDD
	 * @param optionChart est un service pour initialiser le tableau d'options pour chaque chart
	 */
	constructor(private dataService: DataService, private optionChart:OptionChartService) { }

	ngOnInit(): void {
		this.dataService.getBC().subscribe((res: any) => this.bc = res)
		this.dataService.getContributor().subscribe((res: any) => this.contributors = res)
		this.dataService.getBooks().subscribe((res: any) => this.books = res)
	}

	/**
	 * Cette méthode récupére la valeur de l'événement et filtrer le tableau selon cette valeur
	 * 
	 * @param e est un événement de select box
	 */
	onChangeType(e) {
		this.value = e.target.value
		console.log(this.value);
		this.dataObject = []
		this.data1 = []
		this.data2 = []
		if (this.value==='status') this.data = this.dataService.groupBy(this.bc, item => item.status)
		if (this.value==='first_name') this.data = this.dataService.groupBy(this.bc, item => item.first_name)
		if (this.value==='publication_date') this.data = this.dataService.groupBy(this.books, item => item.publication_date)
		if (this.value==='publication_place') this.data = this.dataService.groupBy(this.books, item => item.publication_place)

		
		for (let key of this.data) {
			
			let item = {
				value: key[1].length,
				name: key[0]
			}
			this.dataObject.push(item)
		}
		// this.data.sortList(data1)
		this.dataObject.sort((a, b) => {
			if (a.name > b.name) return 1
			if (a.name < b.name) return -1
			return 0
		})

		this.dataObject.forEach(item => {
			this.data1.push(item.name)
			this.data2.push(item.value)
		})

		console.log(this.dataObject);
		this.dataService.sortList(this.data1)
		console.log(this.data1);
		console.log(this.data2);
		
		

		this.optionPie = this.optionChart.optionPie(this.dataObject)
		console.log(this.optionPie);
		
		this.optionBar = this.optionChart.optionBar(this.data1,this.data2)
		this.optionLine = this.optionChart.optionLine(this.data1,this.data2)
		this.optionPolarRadialBar = this.optionChart.optionPolarRadial(this.dataObject)
		this.optionPolarBar = this.optionChart.optionPolarBar(this.data1,this.data2)
	}

	/**
	 * Cette methode récupère un paramètre et filtre la liste des données selon ce paramètre
	 * 
	 * @param value c'est un paramètre envoyer en cliquant sur le bouton
	 */
	onButtonClick(value) {
		this.value = value
		console.log(value);
		this.dataObject = []
		this.data1 = []
		this.data2 = []
		if (value==='status') this.data = this.dataService.groupBy(this.bc, item => item.status)
		if (value==='first_name') this.data = this.dataService.groupBy(this.bc, item => item.first_name)
		if (value==='publication_date') this.data = this.dataService.groupBy(this.books, item => item.publication_date)
		if (value==='publication_place') this.data = this.dataService.groupBy(this.books, item => item.publication_place)

		
		for (let key of this.data) {
			
			let item = {
				value: key[1].length,
				name: key[0]
			}
			this.dataObject.push(item)
		}
		// this.data.sortList(data1)
		this.dataObject.sort((a, b) => {
			if (a.name > b.name) return 1
			if (a.name < b.name) return -1
			return 0
		})

		this.dataObject.forEach(item => {
			this.data1.push(item.name)
			this.data2.push(item.value)
		})

		console.log(this.dataObject);
		this.dataService.sortList(this.data1)
		console.log(this.data1);
		console.log(this.data2);
		
		

		this.optionPie = this.optionChart.optionPie(this.dataObject)
		console.log(this.optionPie);
		
		this.optionBar = this.optionChart.optionBar(this.data1,this.data2)
		this.optionLine = this.optionChart.optionLine(this.data1,this.data2)
		this.optionPolarRadialBar = this.optionChart.optionPolarRadial(this.dataObject)
		this.optionPolarBar = this.optionChart.optionPolarBar(this.data1,this.data2)
	}

	darkMode(){
		console.log(document.body.style.backgroundColor)
		if (document.querySelector('body').style.background==='#000'){
			document.querySelector('body').style.background='#111'
			document.querySelector('body').style.color='#000'
		}
	}

	lightMode(){
		if (document.querySelector('body').style.background==='#111'){
			document.querySelector('body').style.background='#000'
			document.querySelector('body').style.color='#111'
		}

	}

}

import { Injectable } from '@angular/core';

@Injectable({
  providedIn: 'root'
})
export class OptionChartService {

  constructor() { }

  pie:any
  bar:any
  line:any
  polarRadial:any
  polarBar:any
  radar:any

	optionPie(dataObject) {
        this.pie = {
            legend: {
                data: dataObject
            },
            tooltip: {},
            xAxis: {
                axisLine: {
                    show: false,
                },
            },
            yAxis: {
                axisLine: {
                    show: false,
                },
            },
            series: [
                {
                    name: 'books',
                    data: dataObject,
                    type: 'pie',
                },
            ],
        }
        return this.pie
    }

	optionBar(data1: any, data2: any) {
		this.bar = {
			title: [
				// {
				// 	text: ''
				// }
			],
			yAxis: {
				type: 'category',
				data: data1,
				axisLabel: {
					inside: true,
          			color: 'black',
					rotate: 0,
				},
				// axisTick: {
				// 	show: false,
				//   },
				//   axisLine: {
				// 	show: false,
				//   },
				  z: 10,
			},
			xAxis: {},
			tooltip: {},
			// dataZoom: [
			// 	{
			// 	  type: 'inside',
			// 	},
			//   ],
			series: [
				{
					name: 'books',
					data: data2,
					type: 'bar',
          itemStyle: {
            color: '#83bff6',
          },
          emphasis: {
            itemStyle: {
              color: '#188df0',
            }
          },
				},
			],
		}
		return this.bar
	}

	optionLine(data1: any, data2: any) {
		this.line = {
			title: [
				// {
				// 	text: 'this.title'
				// }
			],
			xAxis: {
				type: 'category',
				data: data1,
				axisLabel: {
					rotate: 90,
				},
			},
			yAxis: {},
			tooltip: {},

			series: [
				{
					name: 'books',
					data: data2,
					type: 'line',
				},
			],
		}
		return this.line
	}

	optionPolarRadial(data: any) {
		this.polarRadial = {
			title: [
				// {
				// 	text: 'this.title'
				// }
			],
			legend: {
				top: 'bottom'
			  },
			  calculable: true,

			tooltip: {
				trigger: 'item',
      			formatter: '{a} <br/>{b} : {c} ({d}%)'
			},

			series: [
				{
					name: 'book',
					data: data,
					radius: [25, 150],
					center: ['50%','50%'],
					type: 'pie',
					roseType: 'area',
					// itemStyle: {
					// 	borderRadius: 8
					//   },
				},
			],
			
  			animation: true
		}
		return this.polarRadial
	}

	optionPolarBar(data1: any, data2: any) {
		this.polarBar = {
			title: [
				// {
				// 	text: 'this.title'
				// }
			],
			polar: {
				radius: [30, '80%']
			},
			angleAxis: {
				max: 4,
				startAngle: 75
			},
			radiusAxis: {
				type: 'category',
				data: data1,
				axisLabel: {
					rotate: 90,
				},
			},
			tooltip: {},

			series: [
				{
					name: 'books',
					data: data2,
					coordinateSystem: 'polar',
					type: 'bar',
				},
			],
		}
		return this.polarBar
	}

	
}
